                        var $cookieFacebook = jQuery('#cookie-facebook').val();
                        var $cookieHotJar = jQuery('#cookie-hotjar').val();
                        var $cookieOlarkChat = jQuery('#cookie-olark').val();
                        var $cookiePardot = jQuery('#cookie-pardot').val();

                        createCookie('cookie-linkedin', $cookieLinkedin);
                        createCookie('cookie-kickfire', $cookieKickfire);
                        createCookie('cookie-lead_forensics', $cookieForensics);
                        createCookie('cookie-indeed-tag', $cookieIndeed);
                        createCookie('cookie-facebook', $cookieFacebook);
                        createCookie('cookie-hotjar', $cookieHotJar);
                        createCookie('cookie-olark', $cookieOlarkChat);

                        // store that we have accepted the cookie policy

                                                jQuery('div#cookie-overlay').hide();

                    });

                    jQuery('div#cookie-overlay a.accept').click(function () {

                        // store that we have accepted the cookie policy
                        createCookie('cookie_policy', true, 365);

                        createCookie('cookie-linkedin', 'On');
                        createCookie('cookie-kickfire', 'On');
                        createCookie('cookie-lead_forensics', 'On');
                        createCookie('cookie-indeed-tag', 'On');
                        createCookie('cookie-facebook', 'On');
                        createCookie('cookie-hotjar', 'On');
                        createCookie('cookie-olark', 'On');
                        createCookie('cookie-pardot', 'On');

                        jQuery('div#cookie-overlay').hide();

                        location.reload();
                    });

                    jQuery('div#cookieModal button.accept-all').click(function () {

                        // store that we have accepted the cookie policy
                        createCookie('cookie_policy', true, 365);

                        // change all dropdown values to yes
                        jQuery('div#cookieModal select option').each(function () {
                            jQuery(this).removeAttr('selected');
                        });

                        // change all dropdown values to yes
                        jQuery('div#cookieModal select option').each(function () {
                            var value = jQuery(this).val();
                            if (value === 'on') {
                                jQuery(this).prop('selected', true);
                            }
                        });

                        // trigger submission
                        jQuery('#btn-cookie-submit').click();
                        jQuery('div#cookie-overlay').hide();
                    });

                    /**
                     * Has to be added due to not being present on some parts of system
                     * Creates a new cookie
                     * @param {string} name
                     * @param {string} value
                     * @param {int} days
                     */
                    function createCookie(name, value, days) {
                        if (days) {
                            var date = new Date();
                            date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
                            var expires = "; expires=" + date.toGMTString();
                        } else
                            var expires = "";
                        document.cookie = name + "=" + value + expires + "; path=/";
                    }

                    /**
                     *  Get cookie by its name
                     * @param {string} name
                     * @returns {unresolved}
                     */
                    function readCookie(name) {
                        var nameEQ = name + "=";
                        var ca = document.cookie.split(';');
                        for (var i = 0; i < ca.length; i++) {
                            var c = ca[i];
                            while (c.charAt(0) == ' ')
                                c = c.substring(1, c.length);
                            if (c.indexOf(nameEQ) == 0)
                                return c.substring(nameEQ.length, c.length);
                        }
                        return null;
                    }

                });

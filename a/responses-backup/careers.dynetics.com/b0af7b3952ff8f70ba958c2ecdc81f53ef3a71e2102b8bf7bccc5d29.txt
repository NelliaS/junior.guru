                    <h3>Position Description</h3>
                    <p>
                        <p>Dynetics Inc., a Wholly Owned Subsidiary of Leidos, is seeking a Radar HWIL Simulation Developers to join a diverse, highly-skilled team to create unique solutions for complex problems involving foreign weapons systems. With offices across the United States engaging in the defense, space, cyber and commercial fields, Dynetics provides responsive, cost-effective intelligence, engineering, scientific and IT solutions.</p><p>Dynetics is seeking talented Radar HWIL Simulation Developers to join a multi-contractor team in creating unique solutions to complex problems in developing models and simulations in support of the Missile Defense Agency in Huntsville Alabama. This role requires a skilled engineer/scientist that understands radars and at least one of the following: algorithm development, real time HWIL software development, signal generation/processing, radar algorithms, and radar performance analysis.</p><p>The individual will analyze radar algorithms and performance and develop/maintain software to be integrated with real time radar signal and data processing equipment to support MDA missions.</p>
                    </p>
                    <h4>Basic Qualifications</h4>
                    <p><p>Candidate must possess a Bachelor&#39;s (or Master&#39;s) Degree in Computer Engineering, Electrical Engineering, Software Engineering, Computer Science, Math, Physics or related STEM Discipline.</p><p>Must have experience in at least one of the following areas:</p><ol><li>algorithm development</li><li>real time HWIL software development</li><li>signal generation/processing</li><li>radar algorithms</li><li>radar performance analysis</li></ol><p>Candidate must be a resident of the North Alabama area or willing to relocate to the Huntsville - North Alabama, AL area.</p></p>
                    <h4>Other Qualifications</h4>
                    <p><p>Radar and signal injection experience and C/C&#43;&#43; programming experience are a plus.</p></p>
                    <h4>Security Requirements</h4>
                    <p><p>Candidate must be a US Citizen and possess (and be able to maintain) a Final Secret Clearance or meet the eligibility to obtain (and maintain) a Secret Clearance. An interim Secret clearance is required to start.</p></p>

                        $emlMisMatch.insertAfter(eml);
                        $emlMisMatch.clone().insertAfter(veml);
                        error = true;
                        $("<div class='alert alert-danger v-failed'>Emails do not match.</div>").insertBefore(submit);
                    }
                    if ( window.hasOwnProperty("cna_duplicateEmail") && cna_duplicateEmail===true ) {
                        $emlFailed.insertAfter(eml);
                        $emlFailed.clone().insertAfter(veml);
                        error = true;
                        $("<div class='alert alert-danger v-failed'>Email already exists.</div>").insertBefore(submit);
                    }
                }
                //-- check passwords
                var $pwdFailed = $("<span class='v-required'> (passwords must match) </span>");
                var pwd        = $(this).find("input[name='password']");
                var vpwd       = $(this).find("input[name='verify_password']");
                if ( pwd.length>0 && vpwd.length>0 ) {
                    if ( pwd.val() != vpwd.val() ) {
                        $pwdFailed.insertAfter(pwd);
                        $pwdFailed.clone().insertAfter(vpwd);
                        error = true;
                        $("<div class='alert alert-danger v-failed'>Passwords do not match.</div>").insertBefore(submit);
                    }
                }
                // console.log( g_createNewAccountForm_captchaHasBeenFilled );
                if ( ! g_createNewAccountForm_captchaHasBeenFilled ) {
                    error = true;
                    errorLabels.push( "Please check the reCAPTCHA box" );
                }
            }

            if(error) { // if error is true;

                //---------------------------------------------------------------------------
                //-- errors here meaning no value entered in required fields
                //---------------------------------------------------------------------------
                var $row = $("<div class='row'></div>");
                var $col = $("<div class='col-sm-12'></div>");
                var $newDiv = $("<div class='alert alert-danger v-failed failed-spot'><h4>Please fill in the required fields</h4></div>");
                var $list = $('<ul></ul>');
                if ( errorLabels && errorLabels.length>0 ) {
                    $.each( errorLabels, function( idx, txt ) {
                        $list.append( "<li>" +txt+ "</li>" )
                    });
                }
                $col.append( $list );
                $row.append( $col );
                $newDiv.append( $row );
                if ( $thisID==='profile-form') {
                    $newDiv.insertBefore( $('#btn_save_profile') );
                } else {
                    $newDiv.insertBefore( submit );
                }
                return false; // stop the form from being submitted.


            } else {

                //---------------------------------------------------------------------------
                //-- Everything has a value but now do PATTERN error checking
                //---------------------------------------------------------------------------
                var patternErrorLabels = [];
                var patternError = false;
                //-- find all patterns
                var $patternFailed = $("<span class='v-required'> (type mismatch) </span>")
                var $withPattern   = $(formClass).find('.required[pattern]');
                if ( $thisID==='profile-form' ) {
                    // $withPattern   = $(formClass).find('.required[data-pattern]');
                    $withPattern   = $(formClass).find('[data-pattern]');
                }
                for(var i = 0; i <= ($withPattern.length - 1);i++) {
                    var item  = $withPattern[i];
                    var $i    = $(item);
                    if ( $i.val().length===0 ) {
                        continue;
                    }
                    var $pat  = $i.attr('pattern');
                    if ( $thisID==='profile-form' ) {
                        $pat = $i.attr('data-pattern');
                    }
                    var match = $i.val().match($pat);
                    if ( !match ) {
                        patternError = true;
                        $i.addClass('bg-required')
                        $patternFailed.insertAfter($i);

                        var eLabel = getErrLabel( $i );
                        if ( eLabel ) {
                            patternErrorLabels.push( eLabel )
                        }
                    }
                }
                if (patternError) {
                    // debugger;

                    var $row = $("<div class='row'></div>");
                    var $col = $("<div class='col-sm-12'></div>");
                    var $newDiv = $("<div class='alert alert-danger v-failed'><h4>Make sure the data entered matches the expected format</h4></div>");
                    var $list = $('<ul></ul>');
                    if ( patternErrorLabels && patternErrorLabels.length>0 ) {
                        $.each( patternErrorLabels, function( idx, txt ) {
                            $list.append( "<li>" +txt+ "</li>" )
                        });
                    }
                    $col.append( $list );
                    $row.append( $col );
                    $newDiv.append( $row );


                    var placeWhere = submit;
                    if ( $thisID==='profile-form') {
                        placeWhere = $('#btn_save_profile');
                    }
                    $newDiv.insertBefore(placeWhere);
                    return false; // stop the form from being submitted.
                }
            }

            if ( $thisID==='profile-form' || $thisID==='the-form' ) {
                return true;
            }
        }

var getCookieByMatch = function(regex) {
    var _cookies = document.cookie.split(/;\s*/),
        _cookieArray = [],
        i;
    for (i = 0; i < _cookies.length; i++) {
        if (_cookies[i].match(regex)) {
            _cookieArray.push(_cookies[i]);
        }
    }
    return _cookieArray;
};

function getUserMarketingID(value) {
    const _regex = /^(.*)\|MCMID\|(\d+)\|(.*)/gm;
    if ((_matchedArray = _regex.exec(value)) !== null && _matchedArray.length >= 3) {
        return _matchedArray[2];
    }
}

function setCookie(cname, cvalue, exdays, path) {
    var expires = "expires=0"
    if (exdays != 0) {
        var d = new Date();
        d.setTime(d.getTime() + (exdays * 24 * 60 * 60 * 1000));
        expires = "expires=" + d.toGMTString();
    }
    document.cookie = cname + "=" + cvalue + "; " + expires + "; path=/;";
}

$.each(getCookieByMatch('AMCV_'), function(index, val) {
    var cookie = val.split("=");
    if (cookie.length >= 2 && typeof Cookies.get('pwc-id') !== 'undefined') {
        var userID = getUserMarketingID(decodeURIComponent(cookie[1]));
        var existingID = Cookies.get('userMID');
        if (typeof existingID === 'undefined' || existingID != userID) {
            var userMCMObj = new Object();
            userMCMObj.userMarketingID = userID;
            userMCMObj.isSetMarketingIDRequest = true;
            $.ajax('/bin/extendedprofile/update', {
                data: JSON.stringify(userMCMObj),
                type: "POST",
                dataType: 'json',
                contentType: 'application/json; charset=utf-8',
                mimeType: 'application/json; charset=utf-8',
                success: function(rawData, status, xhr) {
                    var responseCode = rawData.result.code;
                    if (responseCode === '1001') {
                        setCookie('userMID', userID, 1, '/');
                    }
                },
                error: function(xhr) {
                    console.log("Response is ", xhr);
                }
            });
        }
    }
});

                $(document).on('click.gaEventTrack', '[data-track-event="true"]', function (event) {
                    var $elem = $(this),
                        href = $elem.attr('href'),
                        obj = {},
                        category = $elem.data('eventCategory'),
                        action = $elem.data('eventAction'),
                        label = $elem.data('eventLabel');

                    obj.event = 'gaEvent';

                    if (category !== undefined) {
                        obj.eventCategory = category;
                    }

                    if (action !== undefined) {
                        obj.eventAction = action;
                    }

                    if (label !== undefined) {
                        obj.eventLabel = label;
                    }

                    dataLayer.push(obj);

                    if (href) {
                        event.preventDefault();

                        setTimeout(function () {
                            window.location.href = href;
                        }, 100);

                        return false;
                    }
                });
